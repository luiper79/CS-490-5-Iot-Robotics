int switchState = 0;
void setup() {
  pinMode(3, OUTPUT);
  pinMode(4, OUTPUT);
  pinMode(5, OUTPUT);
  pinMode(2, INPUT);
}

void loop(){
  switchState = digitalRead(2);
  if(switchState == LOW){
  // the button is not pressed

  digitalWrite(3, HIGH); //green LED
  
  
  digitalWrite(4, HIGH); //red LED
 

  digitalWrite(5, HIGH); //yellow LED

  }

  else{

  start:
  digitalWrite(4, HIGH);
  digitalWrite(5, LOW);
  digitalWrite(3, LOW);
  delay(1000);

   
  digitalWrite(4, LOW);
  digitalWrite(3, HIGH);

  delay(750);// wait for a quarter seconds
  digitalWrite(5, HIGH);
  digitalWrite(3, LOW);
  
  delay(500);
  
  digitalWrite(5, LOW);

  goto start;
  }

}
